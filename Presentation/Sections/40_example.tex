\begin{frame}{The Group $S_3$}
    \large
    Its conjugacy classes are

    \vspace{1em}
    \begin{itemize}
        \visible<2->{\item The identity element \hspace*{\fill} $\Cl((1)) = \{(1)\}$}
        \visible<3->{\item The reflections \hspace*{\fill} $\Cl((12)) = \{(13), (23), (12)\}$}
        \visible<4->{\item The rotations \hspace*{\fill} $\Cl((123)) = \{(123), (132)\}$}
    \end{itemize}
\end{frame}

\begin{frame}{The Trivial Representation of $S_3$}
    \large
    Maps all elements to the identity map
    \Large
    \begin{align*}
        \visible<2->{
            \triv: \quad & S_3 \to \GL(\C^1) = \C \setminus \{0\}
        }
        \visible<3->{
            , \\
            & a \mapsto I = 1
        }
    \end{align*}

    \large
    \visible<4->{
        Its character is constant

        \begin{align*}
            \forall a \in S_3: \quad & \chi_{\triv}(a) = 1
        \end{align*}
    }
\end{frame}

\begin{frame}{The Sign Representation of $S_3$}
    \large
    %Represent with a single number, whether the action inverts the direction of ascending numbers in a triangle
    
    %\vspace{1em}
    \visible<2->{Maps all \textbf{inversions} to $-1$, all other elements to 1}

    \begin{align*}
        \visible<3->{
            \sgn: \quad & S_3 \to \GL(\C^1) = \C \setminus \{0\}
        }
        \visible<4->{
            , \\
            & a \mapsto \begin{cases}
                -1 & \text{if } a \in \Cl((12)) \\
                1 & \text{else}
            \end{cases}
        }
    \end{align*}
    
    \visible<5->{
        Its character is

        \begin{align*}
            \chi_{\sgn}(a) = a \mapsto \begin{cases}
                -1 & \text{if } a \in \Cl((12)) \\
                1 & \text{else}
            \end{cases}
        \end{align*}
    }
\end{frame}

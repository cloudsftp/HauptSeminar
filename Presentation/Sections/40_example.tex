\begin{frame}{The Group $S_3$}
    \large
    Its conjugacy classes are

    \vspace{1em}
    \begin{itemize}
        \visible<2->{\item The identity element \hspace*{\fill} $\Cl((1)) = \{(1)\}$}
        \visible<3->{\item The reflections \hspace*{\fill} $\Cl((12)) = \{(13), (23), (12)\}$}
        \visible<4->{\item The rotations \hspace*{\fill} $\Cl((123)) = \{(123), (132)\}$}
    \end{itemize}
\end{frame}

\begin{frame}{The Trivial Representation of $S_3$}
    \large
    Maps all elements to the identity map
    \begin{align*}
        \visible<2->{
            \triv: \quad & S_3 \to \GL(\C^1) = \C \setminus \{0\}
        }
        \visible<3->{
            , \\
            & a \mapsto I = 1
        }
    \end{align*}

    Obviously irreducible (no subspace of $\C$)

    \visible<5->{
        Its character is constant
        \begin{align*}
            \forall a \in S_3: \quad & \chi_{\triv}(a) = 1
        \end{align*}
    }
\end{frame}

\begin{frame}{The Sign Representation of $S_3$}
    \large
    %Represent with a single number, whether the action inverts the direction of ascending numbers in a triangle
    
    %\vspace{1em}
    \visible<2->{Maps all \textbf{inversions} to $-1$, all other elements to 1}
    \begin{align*}
        \visible<3->{
            \sgn: \quad & S_3 \to \GL(\C^1) = \C \setminus \{0\}
        }
        \visible<4->{
            , \\
            & a \mapsto \begin{cases}
                -1 & \text{if } a \in \Cl((12)) \\
                1 & \text{else}
            \end{cases}
        }
    \end{align*}

    Obviously irreducible (no subspace of $\C$)
    
    \visible<5->{
        Its character is
        \begin{align*}
            \chi_{\sgn}(a) = a \mapsto \begin{cases}
                -1 & \text{if } a \in \Cl((12)) \\
                1 & \text{else}
            \end{cases}
        \end{align*}
    }
\end{frame}

\begin{frame}{The Standard Representation of $S_3$}
    \large
    Let $V' = \{\begin{pmatrix}
        x_1 & x_2 & x_3
    \end{pmatrix}^T \in \C^3 | x_1 + x_2 + x_3 = 0\}$
    (set of all points on $\Delta^2$)
    
    \visible<2->{
        This is a subspace of $\C^3$ that is equivalent to $V = \C^2$
    }

    \visible<3->{
        If $\{e_1, e_2, e_3\}$ is a basis of $V'$,
        $\{e_1 - e_2, e_2 - e_3\}$ is a basis of $V$
    }
    
    \vspace{1em}
    \visible<4->{
        So the standard representation is 2-dimensional
        \begin{align*}
            \stand: \quad & S_3 \to \GL(\C^2)
        \end{align*}
    }
    
    \visible<5->{
        For each element $a \in S_3$, the representation $\stand(a)$ is its action on $V$
    }
\end{frame}

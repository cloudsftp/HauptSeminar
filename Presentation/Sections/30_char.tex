\section{Representation of Representations}

\begin{frame}{Combining Representations}
    \Large
    \begin{align*}
        \C^n \oplus \C^m & = \C^{n + m} \\
        \visible<2->{
            A \oplus B & = \begin{pmatrix}
                A & 0 \\
                0 & B
            \end{pmatrix} \\\\
        }
        \visible<3->{
            (\rho \oplus \rho')(a) & = \rho(a) \oplus \rho'(a) & \forall a \in G
        }
    \end{align*}

\end{frame}

\begin{frame}{Simple (irreducible) Representations}
    \large
    $\rho: G \to \GL(V)$ is \textbf{simple},
    
    \visible<2->{
        if there is no non-trivial subspace $U \subset V$ with
    }
    \visible<3->{
        \begin{align*}
            \rho(a) \cdot U & \subset U & \forall a \in G
        \end{align*}
        
        \normalsize
        {\hspace*{\fill} \cite{hein2013}}
    }

\end{frame}

\begin{frame}{Shur's Lemma}
    \large
    $\rho: G \to \GL(V)$ and $\rho': G \to V'$ \textbf{irreducible}

    \vspace{1em}
    \visible<2->{
        $f: V \to V'$ \textbf{intertwinder} of $rho$ and $\rho'$
    }
    
    \vspace{1em}
    \begin{enumerate}
        \visible<3->{
            \item Either $f$ is an \textbf{isomorphism} or $f = 0_{V'}$
        }
        \visible<4->{
            \item If $V = V'$, then $\exists \lambda \in \C:\ f = \lambda \cdot I$
        }
    \end{enumerate}    
    
    \vspace{1em}
    \visible<5->{
        $\implies$ Any representation $\rho$ of a finite group has a \textbf{unique} decomposition
        \begin{align*}
            \rho & = \rho_1^{\oplus a_1} \oplus \rho_2^{\oplus a_2} \oplus \ldots \oplus \rho_k^{\oplus a_k}
        \end{align*}

        \normalsize
        {\hspace*{\fill} \cite{fulton2013}}
    }

\end{frame}
